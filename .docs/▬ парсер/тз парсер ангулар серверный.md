# Техническое задание: Промышленная система веб-парсинга

## 1. Общая архитектура системы

### 1.1 Компоненты системы

```
Angular 19 (UI) → API Gateway → Load Balancer
                                     ↓
                              ASP.NET Core Web API
                                     ↓
Redis Cluster ← → PostgreSQL ← → Parser Engine
     ↓                              ↓
Worker Service ← → Browser Pool ← → Strategy Selection
     ↓                              ↓
Job Scheduler ← → Rate Limiter ← → Data Extractor
```

### 1.2 Технологический стек

**Backend:**

- ASP.NET Core 8.0 + Entity Framework Core
- PostgreSQL (партиционирование, данные)
- Redis Cluster (очереди, кеш, rate limiting)
- Playwright + LightweightJS (V8/JSDOM)
- AngleSharp (HTML парсинг)
- Polly (resilience patterns)
- Prometheus + Grafana (метрики)

**Frontend:**

- Angular 19 (standalone components)
- ng-zorro-antd UI library
- SignalR (real-time updates)

**Infrastructure:**

- Docker + Docker Compose
- NGINX (reverse proxy)

### 1.3 Ключевые принципы

- **Parser Engine** = чистая функция с встроенным rate limiting
- **Strategy fallback**: HTTP → LightweightJS → Playwright (только для сложных случаев)
- **Browser Context** = per site (стабильный fingerprint)
- **Error policies** = конфигурация, не код
- **Rate limiting** = координация через Redis между всеми workers
- **Queue** = Redis Streams (атомарность, retry, DLQ)

```csharp
public interface IRateLimiter

    Task<RateLimitStatus> GetStatusAsync(string domain);

public class TokenBucketRateLimiter : IRateLimiter
{
    private readonly IDatabase _database;

    public async Task<RateLimitResult> TryAcquireAsync(string domain, CancellationToken cancellationToken)
    {
        // Lua script для атомарного token bucket
        const string script = @"
            local key = KEYS[1]
            local capacity = tonumber(ARGV[1])
            local tokens = tonumber(ARGV[2])
            local interval = tonumber(ARGV[3])
            local now = tonumber(ARGV[4])

            local bucket = redis.call('HMGET', key, 'tokens', 'last_refill')
            local current_tokens = tonumber(bucket[1]) or capacity
            local last_refill = tonumber(bucket[2]) or now

            -- Refill tokens based on time passed
            local elapsed = now - last_refill
            local tokens_to_add = math.floor(elapsed / interval)
            current_tokens = math.min(capacity, current_tokens + tokens_to_add)

            if current_tokens >= tokens then
                current_tokens = current_tokens - tokens
                redis.call('HMSET', key, 'tokens', current_tokens, 'last_refill', now)
                redis.call('EXPIRE', key, 3600)
                return {1, current_tokens}
            else
                redis.call('HMSET', key, 'tokens', current_tokens, 'last_refill', now)
                redis.call('EXPIRE', key, 3600)
                return {0, current_tokens}
            end
        ";

        var result = await _database.ScriptEvaluateAsync(script, new RedisKey[] { $"rate_limit:{domain}" },
            capacity, tokensRequested, refillIntervalMs, DateTimeOffset.UtcNow.ToUnixTimeMilliseconds());

        var allowed = (long)result[0] == 1;
        var remainingTokens = (long)result[1];

        return new RateLimitResult
        {
            Allowed = allowed,
            RemainingTokens = (int)remainingTokens,
            RetryAfter = allowed ? TimeSpan.Zero : CalculateRetryAfter(remainingTokens)
        };
    }
}
```

### 3.3 Strategy Selection & Fallback

```csharp
public class HybridParserEngine : IParserEngine
{
    private readonly IParserStrategy[] _strategies;
    private readonly IRateLimiter _rateLimiter;
    private readonly IDataExtractor _extractor;
    private readonly ILogger<HybridParserEngine> _logger;

    public async Task<ParseResult> ParseAsync(ParseTask task, CancellationToken cancellationToken)
    {
        var stopwatch = Stopwatch.StartNew();

        // 1. Load site configuration
        var siteConfig = await GetSiteConfigurationAsync(task.JobId);

        // 2. Apply rate limiting
        var rateLimitResult = await _rateLimiter.TryAcquireAsync(GetDomain(task.Url), cancellationToken);
        if (!rateLimitResult.Allowed)
        {
            await _rateLimiter.WaitForSlotAsync(GetDomain(task.Url), cancellationToken);
        }

        // 3. Select initial strategy
        var strategy = SelectOptimalStrategy(task, siteConfig);
        var strategiesAttempted = new List<string>();

        ContentSnapshot content = null;
        ParseError lastError = null;

        // 4. Execute with fallback
        for (int attempt = 0; attempt < 3; attempt++)
        {
            try
            {
                strategiesAttempted.Add(strategy.Name);
                content = await strategy.LoadContentAsync(task, siteConfig, cancellationToken);

                // 5. Analyze if fallback needed
                var needsFallback = await AnalyzeFallbackNeed(content, task.ExtractionConfig);
                if (!needsFallback)
                    break;

                // 6. Select fallback strategy
                strategy = SelectFallbackStrategy(strategy, task, siteConfig);
                if (strategy == null)
                    break;
            }
            catch (Exception ex)
            {
                lastError = ClassifyError(ex, strategy.Type);
                if (!ShouldRetryWithFallback(lastError))
                    break;

                strategy = SelectFallbackStrategy(strategy, task, siteConfig);
                if (strategy == null)
                    break;
            }
        }

        // 7. Extract data if content loaded successfully
        ExtractedData extractedData = null;
        if (content != null)
        {
            try
            {
                extractedData = await _extractor.ExtractAsync(content, task.ExtractionConfig);
            }
            catch (Exception ex)
            {
                lastError = new ParseError
                {
                    Code = "EXTRACTION_FAILED",
                    Message = ex.Message,
                    Category = ErrorCategory.Parsing,
                    IsRetryable = false
                };
            }
        }

        // 8. Build result
        return new ParseResult
        {
            TaskId = task.Id,
            Url = task.Url,
            Status = content != null && extractedData != null ? ParseStatus.Success : ParseStatus.Failed,
            UsedStrategy = strategy?.Type ?? ParserStrategyType.Http,
            FallbackUsed = strategiesAttempted.Count > 1,
            StrategiesAttempted = strategiesAttempted.ToArray(),
            Data = extractedData,
            Error = lastError,
            TotalDuration = stopwatch.Elapsed,
            RateLimit = new RateLimitInfo
            {
                WasLimited = !rateLimitResult.Allowed,
                RemainingTokens = rateLimitResult.RemainingTokens
            },
            SchemaVersion = "1.0",
            ParsedAt = DateTime.UtcNow
        };
    }

    private IParserStrategy SelectOptimalStrategy(ParseTask task, SiteConfiguration siteConfig)
    {
        // 1. Explicit configuration
        if (siteConfig.PreferredStrategy != ParserStrategyType.Http)
        {
            var explicitStrategy = _strategies.FirstOrDefault(s => s.Type == siteConfig.PreferredStrategy);
            if (explicitStrategy?.CanHandle(task, siteConfig) == true)
                return explicitStrategy;
        }

        // 2. Strategy profile confidence
        if (siteConfig.StrategyProfile.ConfidenceScore > 0.8)
        {
            var profileStrategy = _strategies.FirstOrDefault(s => s.Type == siteConfig.StrategyProfile.PreferredStrategy);
            if (profileStrategy?.CanHandle(task, siteConfig) == true)
                return profileStrategy;
        }

        // 3. Site type heuristics
        var strategyTypes = GetPreferredStrategiesForSiteType(siteConfig.SiteType);
        foreach (var strategyType in strategyTypes)
        {
            var strategy = _strategies.FirstOrDefault(s => s.Type == strategyType);
            if (strategy?.CanHandle(task, siteConfig) == true)
                return strategy;
        }

        // 4. Default HTTP
        return _strategies.FirstOrDefault(s => s.Type == ParserStrategyType.Http);
    }

    private ParserStrategyType[] GetPreferredStrategiesForSiteType(SiteType siteType)
    {
        return siteType switch
        {
            SiteType.Static => new[] { ParserStrategyType.Http },
            SiteType.SPA => new[] { ParserStrategyType.LightweightJS, ParserStrategyType.Playwright },
            SiteType.Protected => new[] { ParserStrategyType.Playwright, ParserStrategyType.LightweightJS },
            SiteType.API => new[] { ParserStrategyType.Http },
            _ => new[] { ParserStrategyType.Http }
        };
    }
}
```

### 4.1 Channel-Based Processing

```csharp
public class ParseWorkerService : BackgroundService
{
    private readonly IParsingQueue _queue;
    private readonly IParserEngine _parserEngine;
    private readonly Channel<ParseTask> _taskChannel;
    private readonly SemaphoreSlim _concurrencySemaphore;
    private readonly string _workerId;
    private readonly int _maxConcurrency;

    public ParseWorkerService(
        IParsingQueue queue,
        IParserEngine parserEngine,
        IConfiguration configuration)
    {
        _queue = queue;
        _parserEngine = parserEngine;
        _workerId = $"worker-{Environment.MachineName}-{Guid.NewGuid():N}";
        _maxConcurrency = configuration.GetValue<int>("Parser:MaxConcurrency", 5);

        var options = new BoundedChannelOptions(_maxConcurrency * 2)
        {
            FullMode = BoundedChannelFullMode.Wait,
            SingleReader = false,
            SingleWriter = true
        };

        _taskChannel = Channel.CreateBounded<ParseTask>(options);
        _concurrencySemaphore = new SemaphoreSlim(_maxConcurrency);
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        // Producer task - reads from Redis Stream
        var producerTask = ProduceTasksAsync(stoppingToken);

        // Consumer tasks - process from channel
        var consumerTasks = Enumerable.Range(0, _maxConcurrency)
            .Select(i => ConsumeTasksAsync($"consumer-{i}", stoppingToken))
            .ToArray();

        try
        {
            await producerTask;
            _taskChannel.Writer.Complete();
            await Task.WhenAll(consumerTasks);
        }
        catch (OperationCanceledException)
        {
            // Graceful shutdown
        }
    }

    private async Task ProduceTasksAsync(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                var task = await _queue.DequeueAsync("workers", _workerId, cancellationToken);
                if (task == null)
                {
                    await Task.Delay(1000, cancellationToken);
                    continue;
                }

                await _taskChannel.Writer.WriteAsync(task, cancellationToken);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error in producer loop");
                await Task.Delay(5000, cancellationToken);
            }
        }
    }

    private async Task ConsumeTasksAsync(string consumerId, CancellationToken cancellationToken)
    {
        await foreach (var task in _taskChannel.Reader.ReadAllAsync(cancellationToken))
        {
            await _concurrencySemaphore.WaitAsync(cancellationToken);

            try
            {
                await ProcessTaskAsync(task, cancellationToken);
            }
            finally
            {
                _concurrencySemaphore.Release();
            }
        }
    }

    private async Task ProcessTaskAsync(ParseTask task, CancellationToken cancellationToken)
    {
        try
        {
            // Parse with Engine
            var result = await _parserEngine.ParseAsync(task, cancellationToken);

            // Save result to database
            await SaveParseResultAsync(result);

            // ACK the message
            await _queue.AckAsync(task.StreamId, task.MessageId);

            // Update job statistics
            await UpdateJobStatsAsync(task.JobId, result.Status == ParseStatus.Success);
        }
        catch (Exception ex)
        {
            await _queue.RetryFailedAsync(task.StreamId, task.MessageId, ex.Message);
        }
    }
}
```

```csharp
public class PlaywrightBrowserPool : IBrowserPool
{
    private readonly SemaphoreSlim _semaphore;
    private readonly ConcurrentQueue<ManagedBrowserContext> _availableContexts;
    private readonly ConcurrentDictionary<string, ManagedBrowserContext> _allContexts;
    private readonly IBrowser _browser;
    private readonly int _maxContexts;

    public async Task<IBrowserContextLease> AcquireContextAsync(long siteId, CancellationToken cancellationToken)
    {
        await _semaphore.WaitAsync(cancellationToken);

        try
        {
            // Try reuse existing context for site
            var existingContext = _availableContexts
                .Where(c => c.SiteId == siteId && c.IsValid)
                .FirstOrDefault();

            if (existingContext != null)
            {
                _availableContexts.TryDequeue(out existingContext);
                existingContext.MarkAsUsed();
                return new BrowserContextLease(existingContext, this);
            }

            // Create new context with site fingerprint
            var fingerprint = await GetSiteFingerprintAsync(siteId);
            var context = await CreateContextWithFingerprintAsync(fingerprint);
            var managedContext = new ManagedBrowserContext(context, siteId);

            _allContexts[managedContext.Id] = managedContext;
            return new BrowserContextLease(managedContext, this);
        }
        catch
        {
            _semaphore.Release();
            throw;
        }
    }

    private async Task<IBrowserContext> CreateContextWithFingerprintAsync(BrowserFingerprint fingerprint)
    {
        var context = await _browser.NewContextAsync(new BrowserNewContextOptions
        {
            UserAgent = fingerprint.UserAgent,
            ViewportSize = ApplyViewportRandomization(fingerprint.Viewport, fingerprint.Randomization),
            Locale = fingerprint.Locale,
            TimezoneId = ApplyTimezoneRandomization(fingerprint.TimezoneId, fingerprint.Randomization),
            ColorScheme = ColorScheme.Light,
            ReducedMotion = ReducedMotion.Reduce
        });

        return context;
    }

    private ViewportSize ApplyViewportRandomization(ViewportSize original, FingerprintRandomization randomization)
    {
        if (!randomization.Enabled)
            return original;

        var jitter = randomization.ViewportJitterRange;
        var random = new Random(randomization.Seed);

        return new ViewportSize
        {
            Width = Math.Max(800, original.Width + random.Next(-jitter, jitter + 1)),
            Height = Math.Max(600, original.Height + random.Next(-jitter, jitter + 1))
        };
    }
}

{

    public string UserAgent { get; set; }
    public ViewportSize Viewport { get; set; }
    public string Locale { get; set; }

    public string[] Languages { get; set; }
    public FingerprintRandomization Randomization { get; set; } = new();

    // Performance tracking
    public DateTime CreatedAt { get; set; }
    public DateTime LastUsedAt { get; set; }
    public int SuccessCount { get; set; }
    public int BlockedCount { get; set; }
```

```csharp
    private readonly ConcurrentDictionary<long, ScheduledJob> _scheduledJobs;

    {
        var schedule = CrontabSchedule.Parse(cronExpression);
        var scheduledJob = new ScheduledJob
        {
            JobId = jobId,
            CronSchedule = schedule,
            CronExpression = cronExpression,
            ScheduleType = ScheduleType.Cron,
            NextRunTime = schedule.GetNextOccurrence(DateTime.UtcNow),
            IsActive = true
        };

        _scheduledJobs.AddOrUpdate(jobId, scheduledJob, (key, old) => scheduledJob);
        await UpdateJobScheduleInDatabase(jobId, cronExpression, ScheduleType.Cron);
    }

    public async Task StartJobAsync(long jobId)
    {
        // Load job from database
        // Generate URLs for parsing
        // Enqueue to Redis Stream
        var urls = await GenerateUrlsForJob(jobId);
        await _queue.EnqueueUrlsAsync(jobId, urls);
    }

    private async void CheckScheduledJobs(object state)
    {
        var now = DateTime.UtcNow;
        var jobsToRun = _scheduledJobs.Values
            .Where(sj => sj.IsActive && sj.NextRunTime <= now)
            .ToArray();

        foreach (var job in jobsToRun)
        {
            try
            {
                await StartJobAsync(job.JobId);
                UpdateNextRunTime(job, now);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error running scheduled job {JobId}", job.JobId);
                job.ErrorCount++;
                if (job.ErrorCount >= 3)
                {
                    job.IsActive = false; // Disable after 3 failures
                }
            }
        }
    }
}
```

```typescript
@Injectable({
  providedIn: "root",
})
export class ParsingService {
  private readonly http = inject(HttpClient);
  private readonly baseUrl = "/api";

  // Auto-refresh jobs every 5 seconds
  private readonly jobsPolling$ = timer(0, 5000).pipe(
    switchMap(() => this.getJobs()),
    share(),
  );

  constructor() {
    this.jobsPolling$.subscribe((jobs) => {
      this.jobsSubject.next(jobs);
    });
  }

  // Sites management
  getSites(): Observable<Site[]> {
    return this.http.get<Site[]>(`${this.baseUrl}/sites`);
  }

  createSite(site: Partial<Site>): Observable<Site> {
    return this.http.post<Site>(`${this.baseUrl}/sites`, site);
  }

  // Jobs management
  getJobs(): Observable<ParseJob[]> {
    return this.http.get<ParseJob[]>(`${this.baseUrl}/jobs`);
  }

  startJob(jobId: number): Observable<void> {
    return this.http.post<void>(`${this.baseUrl}/jobs/${jobId}/start`, {});
  }

  // Real-time updates via SignalR
  connectToHub(): void {
    this.hubConnection = new signalR.HubConnectionBuilder().withUrl("/parsing-hub").build();

    this.hubConnection.start();

    this.hubConnection.on("JobProgress", (jobId: number, progress: JobProgress) => {
      this.updateJobProgress(jobId, progress);
    });
  }
}
```
