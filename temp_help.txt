public async Task<int> SyncIconsByConventionAsync()
{
    var programs = await _repository.GetAllAsync();
    int updatedCount = 0;
    var wwwRoot = _webHostEnvironment.WebRootPath;

    foreach (var program in programs)
    {
        if (string.IsNullOrEmpty(program.Slug)) continue;

        // Конвенция: {первая_буква_slug}/{slug}.png
        char firstChar = char.ToLower(program.Slug[0]);
        string relativePath = $"argregator_icons/programs/icons/{firstChar}/{program.Slug}.png";
        string physicalPath = Path.Combine(wwwRoot, relativePath.Replace("/", "\\"));

        if (File.Exists(physicalPath))
        {
            program.IconPath = relativePath;
            updatedCount++;
        }
    }

    if (updatedCount > 0) await _unitOfWork.SaveChangesAsync();
    return updatedCount;
}`;
